Binding

Description 

di::bind_int<42>::named<my_int>()

bind named with my_int name int parameter to 42 

di::bind<interface, implementation>::named<my_implementation>()

bind named with my_impementation name interface to implementation 

struct my_int { };

class example {
public:
    example(di::named<int, my_int> i)
        : i(i)
    { }

    int i = 0;
};

auto injector = di::make_injector(
    di::bind_int<42>::named<my_int>()
);

assert(42 == injector.create<example>().i);
