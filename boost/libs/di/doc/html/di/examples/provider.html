C++11
#include <boost/di.hpp>

namespace di = boost::di;

struct transaction : di::provider<int> {
    virtual int get() const {
        return next();
    }

    static int& next() {
        static int i = 0;
        return ++i;
    }
};

struct usage {
    usage(int i, std::auto_ptr<di::provider<int> > p) {
        assert(i == 0);
        assert(p->get() == 1);
        assert(p->get() == 2);
    }
};

int main() {
    di::injector<transaction> injector;

    injector.create<usage>();
}
