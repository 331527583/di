Namespace 
boost::di::providers
Examples 
template<class>
class use_heap_provider : public di::config<> {
public:
    auto provider() const noexcept {
        return heap{};
    }
};

// per injector policy
auto injector = di::make_injector<use_heap_provider>();
assert(0 == injector.create<int>());

// global policy
#define BOOST_DI_CFG use_heap_provider
auto injector = di::make_injector();
assert(0 == injector.create<int>());
