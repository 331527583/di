    auto injector = di::make_injector();
    return injector.create<int>();

Create type with bound instance

Asm x86-64 (same as return 42) 

int main() {
    auto injector = di::make_injector(
        di::bind<int>.to(42)
    );

    return injector.create<int>();
}

mov $0x2a,%eax
retq

Create named type

Asm x86-64 (same as return 42) 

auto my_int = []{};

struct c {
    BOOST_DI_INJECT(c
        , (named = my_int) int i)
        : i(i)
    { }

    int i = 0;
};

int main() {
    auto injector = di::make_injector(
        di::bind<int>.named(my_int).to(42)
    );

    return injector.create<c>().i;
}

mov $0x2a,%eax
retq

Create bound interface via module

Asm x86-64 (same as make_unique) 

struct module {
    auto configure() const noexcept {
        return di::make_injector(
            di::bind<i1, impl1>
        );
    }

int main() {
    auto injector = di::make_injector(
        module{}

        unique_ptr<i1>
