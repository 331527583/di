Examples 

class my_config : public di::config {
public:
    auto provider() const noexcept {
        return stack_over_heap{};
    }

    auto policies() const noexcept {
        return di::make_policies(
            [](auto type) { }
          , [](auto type, auto dependency, auto... ctor) { }
          , constructible(std::is_same<int, di::_>{})
        );
    }
};
                                                                                   .

// per injector policy
auto injector = di::make_injector<my_config>();
assert(0 == injector.create<int>());

// global policy
#define BOOST_DI_CFG my_provider
auto injector = di::make_injector();
assert(0 == injector.create<int>());
