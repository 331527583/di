#include <boost/di/config.hpp>

class config : public di::config {
public:
    template<class T>
    auto policies(const T&) const noexcept {
        return di::make_policies(
            [](auto type){ std::cout << typeid(typename decltype(type)::type).name() << std::endl; }
        );
    }
};

    assert(view_.get());

example/modules/controller.cpp 

namespace di = boost::di;

int main() {
   auto injector = di::make_injector(
        module_view{}
      , module_model{}
    );

    return injector.create<controller>().run();
}

example/modules/main.cpp 

namespace di = boost::di;

auto rows = []{};
auto colls = []{};

class model {
public:
    BOOST_DI_INJECT(model, (named = rows) int, (named = colls) int, std::unique_ptr<iboard>);
};

example/modules/model.hpp 

model::model(int rows, int colls, std::unique_ptr<iboard> b) {
    assert(rows == 42);
    assert(colls == 42);
    assert(b.get());
}

namespace di = boost::di;

example/modules/moidule_model.hpp 

namespace di = boost::di;

di::injector<model> module_model::configure() const noexcept {
    return di::make_injector(
        di::bind<iboard, board>
      , di::bind<int>.named(rows).to(42)
      , di::bind<int>.named(colls).to(42)
    );
}
